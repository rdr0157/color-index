let interactiveSelectionHandle = null; window.toggleInteractiveSelection = function (enabled) { if (interactiveSelectionHandle) { interactiveSelectionHandle.remove(); interactiveSelectionHandle = null; } if (!enabled) { setStatus('Interactive parcel selection off.'); return; } deactivateAll(); if (sketch && sketch.state !== 'idle') { sketch.cancel(); } interactiveSelectionHandle = view.on('click', function (event) { if (sketch && sketch.state !== 'idle') return; view.hitTest(event, { include: parcelLayer }).then(function (response) { const result = response.results.find(function (r) { return r.graphic && r.graphic.layer === parcelLayer; }); if (!result || !result.graphic) { setStatus('No parcel selected. Click directly on a parcel.', false); return; } setStatus('Loading selected parcel...'); setActiveParcel(result.graphic, { skipZoom: true }) .then(function () { document.getElementById('loading').classList.add('hidden'); setStatus('Parcel selected. You may now continue drawing or editing.', true); }) .catch(function (err) { console.error(err); setStatus('Unable to load selected parcel.', false); }); }).catch(function (err) { console.error(err); setStatus('Interactive selection failed.', false); }); }); setStatus('Interactive parcel selection on. Click a parcel to load it.'); };
